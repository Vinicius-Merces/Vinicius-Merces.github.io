function loadDespesas() { const user = auth.currentUser; const userRef = db.collection("users").doc(user.uid); const despesasRef = userRef.collection("despesas"); // Exibir todas as despesas cadastradas despesasRef.get().then(querySnapshot => { let despesasList = ""; querySnapshot.forEach(doc => { const despesa = doc.data(); const data = despesa.data; let dataFormatada = "Data inválida"; if (data && data.toDate) { try { dataFormatada = data.toDate().toLocaleDateString(); } catch (error) { console.error("Erro ao formatar a data: ", error); } } let concluidoButton = ''; if (!despesa.concluido) { concluidoButton = ``; } despesasList += `
${despesa.nome}
Valor: R$ ${despesa.valor.toFixed(2)}
Data: ${dataFormatada}
${concluidoButton}
`; }); document.getElementById("despesas-list").innerHTML = despesasList; }).catch(error => { console.error("Erro ao carregar as despesas:", error); }); // Carregar o valor do planejamento userRef.get().then(doc => { if (doc.exists) { const data = doc.data(); const planejamento = data && data.planejamento !== undefined ? data.planejamento : 0; document.getElementById("planejamento").textContent = `R$ ${planejamento.toFixed(2)}`; } else { // Caso o documento não exista, cria o campo planejamento com valor 0 userRef.set({ planejamento: 0 }); document.getElementById("planejamento").textContent = "R$ 0.00"; } }).catch(error => { console.error("Erro ao carregar o planejamento:", error); }); } function loadRelatorios() { const user = auth.currentUser; const periodo = parseInt(document.getElementById("periodo").value); const userRef = db.collection("users").doc(user.uid); const despesasRef = userRef.collection("despesas"); const dataInicio = new Date(); dataInicio.setDate(dataInicio.getDate() - periodo); despesasRef.where("concluido", "==", true) .where("data", ">=", dataInicio) .get().then(querySnapshot => { const despesasConcluidas = []; querySnapshot.forEach(doc => despesasConcluidas.push(doc.data().valor)); const ctx = document.getElementById("grafico").getContext("2d"); if (window.grafico instanceof Chart) { window.grafico.destroy(); } const totalDespesas = despesasConcluidas.reduce((acc, valor) => acc + valor, 0); userRef.get().then(doc => { const data = doc.data(); const planejamento = data && data.planejamento !== undefined ? data.planejamento : 0; window.grafico = new Chart(ctx, { type: "bar", data: { labels: ["Planejamento", "Despesas Concluídas"], datasets: [{ label: "Valor em R$", data: [planejamento, totalDespesas], backgroundColor: ["#4CAF50", "#F44336"] }] }, options: { responsive: true, plugins: { legend: { position: "top", }, title: { display: true, text: "Relatório de Despesas" } } } }); }).catch(error => alert("Erro ao carregar planejamento: " + error.message)); }).catch(error => alert("Erro ao carregar relatório: " + error.message)); }
